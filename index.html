function transliterate(text) { if (!text) return ""; // 1. lowercase let result = text.toLowerCase(); // 2. elimină diacritice (ă â î ș ț etc.) result = result.normalize("NFD").replace(/[\u0300-\u036f]/g, ""); // 3. transliterare chirilică → latină const cyrillicMap = { 'а':'a','б':'b','в':'v','г':'g','д':'d','е':'e','ё':'yo','ж':'zh', 'з':'z','и':'i','й':'y','к':'k','л':'l','м':'m','н':'n','о':'o', 'п':'p','р':'r','с':'s','т':'t','у':'u','ф':'f','х':'kh','ц':'ts', 'ч':'ch','ш':'sh','щ':'sch','ъ':'','ы':'y','ь':'','э':'e','ю':'yu','я':'ya' }; result = result.split('').map(char => cyrillicMap[char] || char).join(''); // 4. elimină tot ce nu e literă sau cifră result = result.replace(/[^a-z0-9]/g, ''); return result; }
